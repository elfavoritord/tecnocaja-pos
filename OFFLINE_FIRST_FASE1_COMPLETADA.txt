### 2️⃣ Backend Helpers (server.js)

8 funciones para gestionar el caché offline:

```javascript
// 1. Inicializar caché para terminal secundaria
await initializeOfflineCache(terminalId, branchId, cashRegisterId, 
                             principalHost, principalBaseUrl);

// 2. Actualizar catálogo de productos
const updated = await updateOfflineProductsCache();
// Result: 2847 productos actualizados

// 3. Actualizar clientes en caché
await updateOfflineClientsCache();

// 4. Actualizar usuarios autorizados
await updateOfflineUsersCache();

// 5. Obtener estado del caché
const status = await getOfflineCacheStatus(terminalId);
// {
//   initialized: true,
//   isOnline: 1,
//   productsCached: 2847,
//   clientsCached: 156,
//   pendingSalesCount: 3,
//   pendingSalesTotalAmount: 1500.00
// }

// 6. Actualizar estado de conexión
await updateTerminalOnlineStatus(terminalId, false, 'offline');

// 7. Registrar evento de sync
await logSyncEvent(terminalId, 'upload', 5, 0, 'ok');

// 8. Generar ID único para venta offline
const offlineId = await generateOfflineInvoiceId(terminalId);
// Result: "a1b2c3d4e5f6g7h8#1#1704067200000"
```

### 3️⃣ Frontend Monitor (js/offline-manager.js)

Clase completamente funcional para monitoreo de conexión:

```javascript
// Crear instancia
const offline = new OfflineManager({
  healthCheckInterval: 2000,      // cada 2 segundos
  syncDebounceDelay: 1000,        // esperar 1s antes de sync
  statusUpdateInterval: 1000      // actualizar UI cada 1s
});

// Inicializar
await offline.initialize();

// Escuchar eventos
offline.on('online', (state) => {
  console.log('✓ Reconectado', state);
  showNotification('Conexión restaurada', 'success');
});

offline.on('offline', (state) => {
  console.log('✕ Desconectado', state);
  showNotification('Modo offline activado', 'warning');
});

offline.on('syncStart', () => {
  showNotification('Sincronizando...', 'info');
});

offline.on('syncComplete', (data) => {
  showNotification('Sincronización completada', 'success');
});

// Obtener estado
const status = offline.getStatusInfo();
// {
//   status: 'offline',
//   label: 'Modo Offline',
//   icon: '✕',
//   color: 'red',
//   pendingCount: 3,
//   pendingTotal: 1500.00
// }

// Forzar sync manual
await offline.forceSync();

// Cancelar venta offline
await offline.cancelPendingSale('a1b2c3d4e5f6g7h8#1#1704067200000');
```

### 4️⃣ Endpoints API

```
GET /api/health
  └─ Verifica disponibilidad del servidor
     Response: { ok: true, app: 'Tecno Caja', version: '2.0.0' }

GET /api/offline/status
  └─ Estado del caché offline
     Response: { initialized, isOnline, productsCached, pendingSalesCount, ... }

POST /api/offline/sync-pending
  └─ Sube ventas offline al servidor (STUB - Fase 4)
     Response: { ok: true, synced, failed, ... }

POST /api/offline/cancel-pending
  └─ Cancela venta offline (STUB - Fase 3)
     Response: { ok: true, cancelled, ... }
```

## 🚀 Cómo Usar - Ejemplos Prácticos

### Backend: Inicializar Caché Después de Login

```javascript
app.post('/api/auth/login', async (req, res) => {
  // ... validar credenciales ...
  
  const user = { id: 123, usuario: 'vendedor1', branch_id: 1, cash_register_id: 1 };
  
  if (isSecondaryTerminal) {
    // Inicializar caché
    await initializeOfflineCache(
      terminalId,                    // "a1b2c3d4e5f6g7h8"
      user.branch_id,               // 1
      user.cash_register_id,        // 1
      principalHost,                // "192.168.1.100"
      principalBaseUrl              // "http://192.168.1.100:3000"
    );
    
    // Cargar datos en paralelo
    await Promise.all([
      updateOfflineProductsCache(),
      updateOfflineClientsCache(),
      updateOfflineUsersCache()
    ]);
  }
  
  return res.json({ ok: true, user });
});
```

### Frontend: Monitorear Conexión

```html
<script src="js/offline-manager.js"></script>
<script>
  const offline = new OfflineManager();
  await offline.initialize();
  
  // Banner de estado
  function updateStatusBanner() {
    const info = offline.getStatusInfo();
    const banner = document.getElementById('status-banner');
    
    banner.innerHTML = `<span style="color: ${info.color}">${info.icon} ${info.label}</span>`;
    
    if (info.pendingCount > 0) {
      banner.innerHTML += ` (${info.pendingCount} ventas pendientes)`;
    }
  }
  
  // Actualizar cuando cambia
  offline.on('statusUpdate', updateStatusBanner);
  offline.on('online', updateStatusBanner);
  offline.on('offline', updateStatusBanner);
  
  updateStatusBanner(); // Mostrar inicial
</script>
```
